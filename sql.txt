CREATE TABLE stackoverflow_questions (
                                         question_id INT PRIMARY KEY,
                                         title VARCHAR(255),
                                         tags VARCHAR(255),
                                         owner_profile_image VARCHAR(255),
                                         owner_account_id INT,
                                         owner_user_type VARCHAR(255),
                                         owner_user_id INT,
                                         owner_link VARCHAR(255),
                                         owner_reputation INT,
                                         owner_display_name VARCHAR(255),
                                         content_license VARCHAR(255),
                                         link VARCHAR(255),
                                         last_activity_date BIGINT,
                                         creation_date BIGINT,
                                         answer_count INT,
                                         score INT,
                                         accepted_answer_id INT,
                                         is_answered BOOLEAN,
                                         view_count INT,
                                         last_edit_date BIGINT,
                                         has_more BOOLEAN,
                                         quota_max INT,
                                         quota_remaining INT,
                                         protected_date BIGINT
                                     );
